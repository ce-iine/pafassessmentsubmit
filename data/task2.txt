db.reviews.insertMany(db.listings.aggregate(
{
    $unwind:"$reviews"
},
{
    $group: {
        _id: '$reviews',
        review: { $first: '$reviews' }
    }
},
{
    $project: {review:1, _id:0}
}).toArray());
